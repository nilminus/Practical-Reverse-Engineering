This function implements a lookup table to translate numbers to single-char strings

char* unk_function(int number){

    switch(number){
        case 6: {
            return "E";
            break;
        }
        case 7: {
            return "D";
            break;
        }
        case 8: {
            return "C";
            break;
        }
        case 9: {
            return "B";
            break;
        }
        default: {
            return "E";
            break;
        }
    }

}
